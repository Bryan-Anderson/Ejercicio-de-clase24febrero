using System;

class Program
{
    static void Main()
    {
        int num, sum = 0;

        Console.WriteLine("Ingrese numeros enteros positivos. Ingresar un n√∫mero negativo para terminar.");

        while (true)
        {
            Console.Write("Ingrese un numero: ");
            string input = Console.ReadLine();

            if (!int.TryParse(input, out num) || num < 0)
            {
                break;
            }

            sum += num;
        }

        Console.WriteLine("La suma de los numeros ingresados es: " + sum);
    }
    }
